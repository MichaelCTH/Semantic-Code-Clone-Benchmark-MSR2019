float str2float (char *s) {
    float result = 0;
    int decimalCount = 0, i = 0, decimalPointLoc = strlen (s);
    for (; s[i] != '\0'; i++) {
        if (s[i] == '.')
            decimalPointLoc = i;
        if (i < decimalPointLoc) {
            result *= 10;
            result += (int) (s[i] + '0');
        }
        else {
            result += (float) (s[i] + '0') / (pow (i -decimalPointLoc, 10));
        }
    }
    return result;
}
