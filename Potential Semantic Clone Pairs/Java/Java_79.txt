        if (eventType == XmlPullParser.START_TAG) {

public int solution (int [] A) {
    int [] sumLeft = new int [A.length];
    int [] sumRight = new int [A.length];
    sumLeft [0] = A [0];
    sumRight [A.length - 1] = A [A.length - 1];
    for (int i = 1;
    i < A.length; i ++) {
        sumLeft [i] = A [i] + sumLeft [i - 1];
    }
    for (int i = A.length - 2;
    i >= 0; i --) {
        sumRight [i] = sumRight [i + 1] + A [i];
    }
    for (int i = 0;
    i < A.length; i ++) {
        if (sumLeft [i] == sumRight [i]) {
            return i;
        }
    }
    return - 1;
}

public static int PermSolution (int [] A) {
    int len = A.length;
    if (len == 1) return A [0] == 1 ? 1 : 0;

    BitSet set = new BitSet (len + 2);
    for (int i = 0;
    i < len; i ++) {
        if (A [i] > len || set.get (A [i])) return 0;

        set.set (A [i]);
    }
    return set.nextClearBit (1) == (len + 1) ? 1 : 0;

public static boolean [] PrimeArray (int [] arr) {
    boolean prime [] = new boolean [arr.length];
    for (int j = 0;
    j < arr.length; j ++) {
        for (int i = 2;
        i <= Math.sqrt (arr [j]); i ++) {
            if (arr [j] % i == 0) {
                prime [j] = false;
                break;
            } else {
                prime [j] = true;
            }
        }
    }
    return prime;
}

private File extractFile (String sourcePath, String sourceCrc, File extractedFile) throws IOException {
    String extractedCrc = null;
    if (extractedFile.exists ()) {
        try {
            extractedCrc = crc (new FileInputStream (extractedFile));
        } catch (FileNotFoundException ignored) {
        }
    }
    if (extractedCrc == null || ! extractedCrc.equals (sourceCrc)) {
        try {
            InputStream input = readFile (sourcePath);
            extractedFile.getParentFile ().mkdirs ();
            FileOutputStream output = new FileOutputStream (extractedFile);
            byte [] buffer = new byte [4096];
            while (true) {
                int length = input.read (buffer);
                if (length == - 1) break;

                output.write (buffer, 0, length);
            }
            input.close ();
            output.close ();
        } catch (IOException ex) {
            throw new RuntimeException ("Error extracting file: " + sourcePath + "\nTo: " + extractedFile.getAbsolutePath (), ex);
        }
    }
    return extractedFile;
}
