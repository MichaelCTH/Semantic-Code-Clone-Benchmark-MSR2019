private static boolean isDescending (List < Integer > getallen, int start, int end) {
    if (end - start <= 1) return true;

    if (end - start == 2) {
        return getallen.get (start) > getallen.get (start + 1);
    }
    int middle = (start + end - 1) / 2 + 1;
    return (getallen.get (middle - 1) > getallen.get (middle)) && isDescending (getallen, start, middle) && isDescending (getallen, middle, end);

public Polynomial2 divides (Polynomial2 b) {
    Polynomial2 a = this;
    if ((b.deg == 0) && (b.coef [0] == 0)) throw new RuntimeException ("Divide by zero polynomial");

    if (a.deg < b.deg) return new Polynomial2 (0, 0);

    int coefficient = a.coef [a.deg] / (b.coef [b.deg]);
    int exponent = a.deg - b.deg;
    Polynomial2 c = new Polynomial2 (coefficient, exponent);
    return c.plus ((a.minus (b.times (c)).divides (b)));
