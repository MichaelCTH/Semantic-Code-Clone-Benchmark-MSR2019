    char s [8];
    divisor = 100000;

        if (i == 3) {
            s[i] = ',';
            continue;

        if (divisor == 1) {

            break;

void delete (int value, struct link **head) {
    if (!head) {
        printf ("error");
        return;
    }
    while (*head) {
        if ((*head)->data == value) {
            struct link *tmp = *head;

            free (tmp);
        }
        else {
            head = &(*head)->next;
        }
    }
}

int delete (struct link **head, int value) {
    int success = 0;
    if (head == NULL)
        return success;
    struct link *target = *head;
    struct link *prev = NULL;
    while (target != NULL & target->data != value) {
        prev = target;
        target = target->next;
    }
    success = target != NULL;
    if (success) {
        if (prev == NULL) {
            *head = target->next;
        }
        else {
            prev->next = target->next;
        }
        free (target);
    }
    return success;
}

            if (ch == '\n') {
                status = none;
                putchar (ch);
