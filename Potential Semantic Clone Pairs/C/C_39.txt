        ++size;

int bitpatSearch (unsigned int source, unsigned int pattern, unsigned int n) {
    unsigned int mask;
    unsigned int tmp;
    int i;
    if (n == 0) {
        return 0;
    }
    if (n == CHAR_BIT * sizeof (source)) {
        if (source == pattern) {
            return 0;
        }
        return NO_MATCH;
    }
    if (n > CHAR_BIT * sizeof (source)) {
        return ARGUMENT_ERROR;
    }
    mask = ~((~0u) << n);
    pattern &= mask;
    for (i = 0; i <= CHAR_BIT * sizeof (source) - n; i++) {
        tmp = (source >> i) & mask;
        if (tmp == pattern) {
            return i;
        }
    }
    return NO_MATCH;
