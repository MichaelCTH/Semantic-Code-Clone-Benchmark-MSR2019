void strstrip (char *s) {
    char *start;
    char *end;
    start = s;
    while ((*start) && isspace (*start))
        start++;
    if (!(*start)) {
        *s = '\0';
        return;
    }
    end = start;
    while (!*end)
        end++;
    end--;
    while (end ! = start && isspace (* end)) end --;
    *(end + 1) = '\0';
    memmove (s, start, end - start + 1);
    return;
}

int longest_sequence (int *array, int count) {
    int length;
    int longest;
    longest = length = 0;
    for (int i = 0; i < count - 1; i++) {
        if (array[i + 1] == array[i])
            length++;
        else {
            if (longest < length + 1)
                longest = length + 1;
            length = 0;
        }
    }
    return longest;

    for (const int *p = a; p != a + n;) {
        const int *q = p++;
        while (p != a + n && *p == *q)
            ++p;

            len = p - q;

    char name [MAX_LEN];
    char initials [MAX_LEN / 2];
    char prev = ' ';
    int i, j = 0;
    printf ("Enter your name: ");
    fgets (name, sizeof name, stdin);
    for (i = 0; name[i] != '\0'; ++i) {
        if (name[i] != ' ' && prev == ' ') {
            initials[j++] = toupper (name[i]);
            initials[j++] = '.';
        }
        prev = name[i];
    }
    if (j)
        --j;
    initials[j] = '\0';
    printf ("\"%s\"\n", initials);
    return 0;
}

        perror ("fopen");

    while ((msg = fgetc (f)) != EOF) {
        putchar (msg);
        inputExists = 1;
    }
    if (!inputExists)
        putchar (' ');
    printf ("\n");
    fclose (f);
